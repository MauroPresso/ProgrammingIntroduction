import sqlite3

class conexionBD:
    #Se activa cada vez que iniciamos una instancia
    #cada vez que se cree una instancia se establezca la conexi√≥n con la BDD
    #defino el connect y el cursor
    def __init__ (self):
        self.con=sqlite3.connect("Base")
        self.cursor=self.con.cursor()
        
    #aca cierro la conexion y el cursor
    def cerrar(self):
        self.con.commit()
        self.con.close()
